// Interactive Dashboard
class StudyTracker {
    constructor() {
        this.data = [];
        this.init();
    }

    // Initialize Dashboard
    init() {
        this.loadData();
        this.setupEventListeners();
        this.renderCharts();
    }

    // Load Data from CSV
    async loadData() {
        try {
            const response = await fetch('/api/data');
            this.data = await response.json();
            this.updateUI();
        } catch (error) {
            console.error('Error loading data:', error);
        }
    }

    // Render Charts
    renderCharts() {
        const ctx = document.getElementById('studyChart').getContext('2d');
        new Chart(ctx, {
            type: 'bar',
            data: this.getChartData(),
            options: this.getChartOptions()
        });
    }

    // Update UI
    updateUI() {
        document.getElementById('totalHours').textContent = this.getTotalHours();
        document.getElementById('studyDays').textContent = this.getStudyDays();
        document.getElementById('averageHours').textContent = this.getAverageHours();
    }
}

React Components
jsx
// Dashboard Component
import React, { useState, useEffect } from 'react';
import { Bar } from 'react-chartjs-2';

const Dashboard = () => {
    const [data, setData] = useState([]);
    const [stats, setStats] = useState({});

    useEffect(() => {
        fetchData();
    }, []);

    const fetchData = async () => {
        try {
            const response = await fetch('/api/data');
            const jsonData = await response.json();
            setData(jsonData);
            calculateStats(jsonData);
        } catch (error) {
            console.error('Error:', error);
        }
    };

    const calculateStats = (data) => {
        const total = data.reduce((sum, item) => sum + item.hours, 0);
        const days = new Set(data.map(item => item.date)).size;
        const average = total / days;
        setStats({ total, days, average });
    };

    return (
        <div className="dashboard">
            <div className="stats-grid">
                <div className="stat-card">
                    <h3>Total Hours</h3>
                    <p className="stat-value">{stats.total}h</p>
                </div>
                <div className="stat-card">
                    <h3>Study Days</h3>
                    <p className="stat-value">{stats.days}</p>
                </div>
                <div className="stat-card">
                    <h3>Average/Day</h3>
                    <p className="stat-value">{stats.average}h</p>
                </div>
            </div>
            <div className="chart-container">
                <Bar data={chartData} options={chartOptions} />
            </div>
        </div>

import pandas as pd
import os
from datetime import datetime, timedelta

class StudyTracker:
    def __init__(self):
        self.csv_file = "study_data.csv"
        self.subjects = ["Math", "Physics", "Digital Logic", "Microprocessor", "Python", "DSA"]
        self.initialize_csv()
    
    def initialize_csv(self):
        """Create CSV file if it doesn't exist"""
        if not os.path.exists(self.csv_file):
            df = pd.DataFrame(columns=["Date", "Subject", "Hours"])
            df.to_csv(self.csv_file, index=False)
            print("✅ CSV file initialized!")
    
    def load_data(self):
        """Load data from CSV"""
        try:
            return pd.read_csv(self.csv_file)
        except:
            return pd.DataFrame(columns=["Date", "Subject", "Hours"])
    
    def save_data(self, df):
        """Save data to CSV"""
        df.to_csv(self.csv_file, index=False)
    
    def add_record(self):
        """Add new study record"""
        print("\n" + "="*50)
        print("📚 ADD STUDY RECORD")
        print("="*50)
        
        # Date selection
        print("\nDate Option:")
        print("1. Today (Auto)")
        print("2. Manual Entry (YYYY-MM-DD)")
        choice = input("Choose (1/2): ")
        
        if choice == "1":
            date = datetime.now().strftime("%Y-%m-%d")
        else:
            date = input("Enter Date (YYYY-MM-DD): ")
            try:
                datetime.strptime(date, "%Y-%m-%d")
            except:
                print("❌ Invalid date format!")
                return
        
        # Subject selection
        print("\nChoose Subject:")
        for i, subject in enumerate(self.subjects, 1):
            print(f"{i}. {subject}")
        subj_choice = int(input("Enter Choice (1-6): ")) - 1
        subject = self.subjects[subj_choice]
        
        # Hours input
        hours = float(input("Enter Hours: "))
        if hours <= 0:
            print("❌ Hours must be positive!")
            return
        
        # Save record
        df = self.load_data()
        new_record = pd.DataFrame({
            "Date": [date],
            "Subject": [subject],
            "Hours": [hours]
        })
        df = pd.concat([df, new_record], ignore_index=True)
        self.save_data(df)
        
        print(f"\n✅ Record Added!")
        print(f"📅 {date} | 📖 {subject} | ⏱️ {hours}h")
